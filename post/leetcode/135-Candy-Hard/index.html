class Solution:
    def candy(self, ratings: List[int]) -> int:
        n = len(ratings)
        c = [-1] * n
        for i in range(n):
            res = []
            if i > 0 and ratings[i - 1] < ratings[i]:
                res.append(i - 1)
            if i < n - 1 and ratings[i + 1] < ratings[i]:
                res.append(i + 1)
            c[i] = 1 if len(res) == 0 else res

        def cal(idx):
            if type(c[idx]) == int:
                return c[idx]
            res = []
            for x in c[idx]:
                res.append(cal(x) + 1)
            c[idx] = max(res)
            return c[idx]

        return sum(c)

class Solution:
    def candy(self, ratings: List[int]) -> int:
        n = len(ratings)
        left, right = [1] * n, [1] * n

        for i in range(n):
            if i > 0 and ratings[i - 1] < ratings[i]:
                left[i] = left[i - 1] + 1

        for i in range(n - 1, -1, -1):
            if i < n - 1 and ratings[i + 1] < ratings[i]:
                right[i] = right[i + 1] + 1

        ans = 0
        for i in range(n):
            ans += max(left[i], right[i])

        return ans
