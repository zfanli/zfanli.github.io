class Solution:
    def maxDepth(self, root: TreeNode) -> int:
        ans = 0

        def traverse(node, depth):
            if not node:
                return
            nonlocal ans
            ans = max(ans, depth)
            traverse(node.left, depth + 1)
            traverse(node.right, depth + 1)

        traverse(root, 1)
        return ans

class Solution:
    def maxDepth(self, root: TreeNode) -> int:

        def traverse(node):
            if not node:
                return 0
            return 1 + max(traverse(node.left),
                           traverse(node.right))

        return traverse(root)
