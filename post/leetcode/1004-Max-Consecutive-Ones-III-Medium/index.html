class Solution:
    def longestOnes(self, nums: List[int], k: int) -> int:
        ans, i, zeros = 0, 0, []
        for j in range(len(nums)):
            if nums[j] == 0:
                zeros.append(j)
                if len(zeros) > k:
                    i = zeros.pop(0) + 1
            ans = max(ans, j - i + 1)
        return ans

class Solution:
    def longestOnes(self, nums: List[int], k: int) -> int:
        ans, i = 0, 0
        for j in range(len(nums)):
            if nums[j] == 0:
                if k == 0:
                    while nums[i] != 0:
                        i += 1
                    i += 1
                else:
                    k -= 1
            ans = max(ans, j - i + 1)
        return ans
