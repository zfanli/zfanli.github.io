class Solution:
    def superpalindromesInRange(self, left: str, right: str) -> int:
        L, R = int(left), int(right)
        ans = 0

        for i in self.table:
            if i < L:
                continue
            elif i <= R:
                ans += 1
            else:
                break

        return ans
