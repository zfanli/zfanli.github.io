class Solution:
    def numSubmatrixSumTarget(self, matrix: List[List[int]], target: int) -> int:
        rows, cols, ans = len(matrix), len(matrix[0]), 0

        for sr in range(rows):
            colsum = [0] * cols
            for er in range(sr, rows):
                d, cur = {0:1}, 0
                for c in range(cols):
                    colsum[c] += matrix[er][c]
                    cur += colsum[c]
                    ans += d.get(cur-target, 0)
                    d[cur] = d.get(cur, 0) + 1

        return ans
