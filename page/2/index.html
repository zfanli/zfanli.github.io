    def criticalConnections(self, n: int, connections: List[List[int]]) -> List[List[int]]:
        # make adjacency-list of connections
        adj = defaultdict(list)
        for u, v in connections:
            adj[u].append(v)
            adj[v].append(u)

        # define dfs
        def dfs(u, prev, id, ids, ans):
            ids[u] = id
            for v in adj[u]:
                if v == prev:
                    continue
                elif ids[v] == -1:
                    ids[u] = min(ids[u], dfs(v, u, id+1, ids, ans))
                else:
                    ids[u] = min(ids[u], ids[v])

            if id != 0 and ids[u] == id:
                ans.append([u, prev])

            return ids[u]

        ids = [-1] * n
        dfs(0, -1, 0, ids, ans)

        return ans

class Solution:
    def numSubmatrixSumTarget(self, matrix: List[List[int]], target: int) -> int:
        rows, cols, ans = len(matrix), len(matrix[0]), 0

        for sr in range(rows):
            colsum = [0] * cols
            for er in range(sr, rows):
                d, cur = {0:1}, 0

                    colsum[c] += matrix[er][c]
                    cur += colsum[c]

                    d[cur] = d.get(cur, 0) + 1

        return ans

class Solution:
    def longestStrChain(self, words: List[str]) -> int:
        memo = {}
        for w in words:
            memo[w] = -1

        def dfs(w):
            if memo[w] != -1:
                return memo[w]
            c = 0
            for i in range(len(w)):
                _w = w[:i] + w[i+1:]
                if _w in memo:
                    c = max(c, dfs(_w))
            memo[w] = c + 1
            return memo[w]

        for w in words:
            dfs(w)

        # print(memo)
        return max(memo.values())

class Solution:
    def longestStrChain(self, words: List[str]) -> int:
        words.sort(key=lambda x: len(x))
        dp = {}
        for w in words:
            c = 1
            for i in range(len(w)):
