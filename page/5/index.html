class Solution:
    def evalRPN(self, tokens: List[str]) -> int:
        stack = []

        for t in tokens:
            if t not in {"+", "-", "*", "/"}:
                stack.append(int(t))
            else:
                right, left = stack.pop(), stack.pop()
                # print(left, t, right)
                if t == '+':
                    stack.append(left + right)
                elif t == '-':
                    stack.append(left - right)
                elif t == '*':
                    stack.append(left * right)
                else:
                    stack.append(trunc(left / right))

        return stack[0]

            if not curr:

    def candy(self, ratings: List[int]) -> int:

        c = [-1] * n

                return c[idx]
            res = []
            for x in c[idx]:
                res.append(cal(x) + 1)
            c[idx] = max(res)
            return c[idx]

class Solution:
    def candy(self, ratings: List[int]) -> int:
        n = len(ratings)
        left, right = [1] * n, [1] * n

        for i in range(n):
            if i > 0 and ratings[i - 1] < ratings[i]:
                left[i] = left[i - 1] + 1

        for i in range(n - 1, -1, -1):
            if i < n - 1 and ratings[i + 1] < ratings[i]:
                right[i] = right[i + 1] + 1

        ans = 0
        for i in range(n):
            ans += max(left[i], right[i])

        return ans

class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
        ans, nums = 0, set(nums)
        for n in nums:
            if n - 1 not in nums:
                count = 1
                while n + 1 in nums:
                    n, count = n + 1, count + 1
                ans = max(ans, count)
        return ans
